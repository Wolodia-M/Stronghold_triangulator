=----------ANGLE----------=
C - angle, towards positive Z, -180...180;
=---------DATAIN----------=
in1: Z1, X1, C1
in2: Z2, X2, C2
=----------DATAOUT----------=
Z3, X3
=----------ANGLEDECODE----------=
f - angle 0...360
if (C == 0) f = 0;
if (C < 0) f = abs(C);
if (C > 0) f = 360 - C;
